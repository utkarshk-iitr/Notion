We were given with two audios and ciphertext , these audios were actually morse code. Audio 1 was iv and Audio 2 was key in morse code so decoding it using website https://databorder.com/transfer/morse-sound-receiver we get the result with some errors.

from Crypto.Cipher import AES
from Crypto.Random import get_random_bytes
from Crypto.Util.number import bytes_to_long

def decrypt_aes_cbc(key, iv, ciphertext):
    cipher = AES.new(key, AES.MODE_CBC, iv)
    plaintext = cipher.decrypt(ciphertext)
    return plaintext

def main():
    key = 'YOUGOTTHEKEYNJOY'.lower().encode()
    iv = "0x000102030405060708090a0b0c0d0e0f"[2:]
    ciphertext = "69d5deb91a001151db5d98231574a51779acd1a84b9338a6750697c0af7e4591"
    ciphertext = bytes.fromhex(ciphertext)
    iv = bytes.fromhex(iv)
    
    plaintext = decrypt_aes_cbc(key, iv, ciphertext)
    print("Decrypted plaintext:", plaintext)

main()

Crypto / Echoes of Encryption

We got ciphertext and encrypt.py file seeing the encryption script we just have to find the seed so that exact random leters could be generated. Now i was unable to guess the seed from the given description so i decided to do bruteforce it.
Now since we know that flag starts with
0CTF{ so i XORed it with ciphertext to get the starting random letters, then this script will decrypt this.

from pwn import xor
import random
import string

data = "0CTF{".encode()
ciphertext = bytes.fromhex("5e04610a22042638723c571e1a5436142764061f39176b4414204636251072220a35583a60234d2d28082b")
key = b""

for i in range(5):
    key += xor(data[i],ciphertext[i])

key = key.decode()

def findkey(l, seed):
    random.seed(seed)    
    allowed_chars = string.ascii_letters + string.digits
    key = ''.join(random.choices(allowed_chars, k=l))
    return key

seed = -1
for i in range(10000000000000):
    k = findkey(len(ciphertext),i)
    if k[:5] == key:
        seed = i
        print(i)
        break

k = findkey(seed)
plaintext = ''
for i in range(len(ciphertext)):
    decrypt = chr(ord(ciphertext[i]) ^ ord(k[i]))
    plaintext += decrypt
print(plaintext)
