Franklin Reiter Attack RSA

When M1 = a1 m + b1 and M2 = a2 m + b2 and both are encrypt by same rsa pair.

from sage.all import *
from Crypto.Util.number import long_to_bytes
import codecs
# All the variable names mean the same as mentioned in the explanation
# For eg, a,b are the values in the function f = ax + b

preparse("P.<X> = PolynomialRing(Zmod(N))")
def gcd(a, b): 
    while b:
        a, b = b, a % b
    return a.monic()

def franklinreiter(C1, C2, e, N, a1, b1, a2, b2):
    P.<X> = PolynomialRing(Zmod(N))
    g1 = (a1*X + b1)^e - C1
    g2 = (a2*X + b2)^e - C2
    result = -gcd(g1, g2).coefficients()[0]
    return long_to_bytes(result)
